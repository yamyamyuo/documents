	public int removeDuplicates(int[] A) {
    	if (A == null ||A.length == 0) {
    		return 0;
    	}
        int length = 1;
        for (int i = 1; i < A.length; i++) {
        	if (A[i]!=A[i-1]) {
        		A[length] = A[i];
        		length++;
        	}
        }
        return length;
    }
